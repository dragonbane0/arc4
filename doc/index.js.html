/**
 * generate ksa
 * 
 * @function gKsa
 * @param {Array} key - user key
 * @return {Array}
 */
function gKsa(key) {

    var j = 0;
    var s = gSbox();
    var ke = key;
    var len = ke.length;
    for (var i = 0; i < 256; i++) {
        j = (j + s[i] + ke[i % len]) % 256;
        s[j] = [s[i],s[i] = s[j]][0];
    }
    return s;
}

/**
 * body cipher
 * 
 * @function body
 * @param {String|Array|Buffer} inp - input
 * @param {String|Array|Buffer} res - response
 * @param {Array} ksa - ksa box
 * @return {String|Array|Buffer}
 */
function body(inp,res,ksa) {

    var i = 0, j = 0;
    if (typeof (res) == 'string') {
        for (var y = 0, l = inp.length; y < l; y++) {
            i = (i + 1) % 256;
            j = (j + ksa[i]) % 256;
            ksa[j] = [ksa[i],ksa[i] = ksa[j]][0];
            res += String.fromCharCode(inp.charCodeAt(y)
                    ^ ksa[(ksa[i] + ksa[j]) % 256]);
        }
    } else {
        for (var y = 0, l = inp.length; y < l; y++) {
            i = (i + 1) % 256;
            j = (j + ksa[i]) % 256;
            ksa[j] = [ksa[i],ksa[i] = ksa[j]][0];
            res[y] = inp[y] ^ ksa[(ksa[i] + ksa[j]) % 256];
        }
    }
    return res;
}

/**
 * body cipher RC4A
 * 
 * @function bodyRC4A
 * @param {String|Array|Buffer} inp - input
 * @param {String|Array|Buffer} res - response
 * @param {Array} ksa - ksa box
 * @return {String|Array|Buffer}
 */
function bodyRC4A(inp,res,ksa) {

    var i = 0, j1 = 0, j2 = 0;
    var s2 = ksa.slice();
    if (typeof (res) == 'string') {
        for (var y = 0, l = inp.length; y < l; y++) {
            i = (i + 1) % 256;
            j1 = (j1 + ksa[i]) % 256;
            ksa[j1] = [ksa[i],ksa[i] = ksa[j1]][0];
            res += String.fromCharCode(inp.charCodeAt(y)
                    ^ s2[(ksa[i] + ksa[j1]) % 256]);
            y++;
            j2 = (j2 + s2[i]) % 256;
            s2[j2] = [s2[i],s2[i] = s2[j2]][0];
            res += String.fromCharCode(inp.charCodeAt(y)
                    ^ ksa[(s2[i] + s2[j2]) % 256]);
        }
    } else {
        for (var y = 0, l = inp.length; y < l; y++) {
            i = (i + 1) % 256;
            j1 = (j1 + ksa[i]) % 256;
            ksa[j1] = [ksa[i],ksa[i] = ksa[j1]][0];
            res[y] = inp[y] ^ s2[(ksa[i] + ksa[j1]) % 256];
            y++;
            j2 = (j2 + s2[i]) % 256;
            s2[j2] = [s2[i],s2[i] = s2[j2]][0];
            res[y] = inp[y] ^ ksa[(s2[i] + s2[j1]) % 256];
        }
    }
    return res;
}

/**
 * body cipher VMPC
 * 
 * @function bodyVMPC
 * @param {String|Array|Buffer} inp - input
 * @param {String|Array|Buffer} res - response
 * @param {Array} ksa - ksa box
 * @return {String|Array|Buffer}
 */
function bodyVMPC(inp,res,ksa) {

    var i = 0, j = 0;
    var a, b;
    if (typeof (res) == 'string') {
        for (var y = 0, l = inp.length; y < l; y++) {
            a = ksa[i];
            j = ksa[(j + a) % 256];
            b = ksa[j];
            res += String.fromCharCode(inp.charCodeAt(y) ^ ksa[ksa[b] + 1]);
            ksa[j] = [a,ksa[i] = b][0];
            i = (i + 1) % 256;
        }
    } else {
        for (var y = 0, l = inp.length; y < l; y++) {
            a = ksa[i];
            j = ksa[(j + a) % 256];
            b = ksa[j];
            res[y] = inp[y] ^ ksa[ksa[b] + 1];
            ksa[j] = [a,ksa[i] = b][0];
            i = (i + 1) % 256;
        }
    }
    return res;
}

/**
 * body cipher RC4P
 * 
 * @function bodyRC4P
 * @param {String|Array|Buffer} inp - input
 * @param {String|Array|Buffer} res - response
 * @param {Array} ksa - ksa box
 * @return {String|Array|Buffer}
 */
function bodyRC4P(inp,res,ksa) {

    var i = 0, j = 0;
    var a, b, c;
    if (typeof (res) == 'string') {
        for (var y = 0, l = inp.length; y < l; y++) {
            i = (i + 1) % 256;
            a = ksa[i];
            j = ksa[(j + a) % 256];
            b = ksa[j];
            ksa[j] = [a,ksa[i] = b][0];
            c = (ksa[i << 5 ^ j >> 3] + ksa[j << 5 ^ i >> 3]) % 256;
            res += String.fromCharCode(inp.charCodeAt(y)
                    ^ (ksa[a + b] + ksa[c ^ 0xAA]) ^ ksa[j + b]);
        }
    } else {
        for (var y = 0, l = inp.length; y < l; y++) {
            i = (i + 1) % 256;
            a = ksa[i];
            j = ksa[(j + a) % 256];
            b = ksa[j];
            ksa[j] = [a,ksa[i] = b][0];
            c = (ksa[i << 5 ^ j >> 3] + ksa[j << 5 ^ i >> 3]) % 256;
            res[y] = inp[y] ^ (ksa[a + b] + ksa[c ^ 0xAA]) ^ ksa[j + b];
        }
    }
    return res;
}

/**
 * RC4 class
 * 
 * @class RC4
 * @param {String|Array|Buffer} key - user key
 * @return {Object}
 */
function RC4(key) {

    this.key;
    this.change(key);
}
/**
 * change user key
 * 
 * @function change
 * @param {String|Array|Buffer} key - user key
 */
RC4.prototype.change = function(key) {

    this.key = new Array(key.legth);
    if (Array.isArray(key)) {
        this.key = key;
    } else if (typeof (key) == 'string' || Buffer.isBuffer(key)) {
        key = new Buffer(key);
        for (var i = 0, ii = key.length; i < ii; i++) {
            this.key[i] = key[i];
        }
    } else {
        throw new Error('Invalid data');
    }
    return;
};
/**
 * RC4 string code
 * 
 * @function codeString
 * @param {String} str - data

RC4.prototype.codeString = function(str) {

    return body(str,'',gKsa(this.key));

/**
 * RC4 mixed code. Alias for codeString or codeByte

    if (typeof (boh) == 'string') {

        throw new Error('Invalid data');
    }
    return;
};
